_columnMetaData Object 
						{
						field : '',
						index : '',//数据索引，默认==field
						title : '',
						width : 120,//默认的列宽,可设置百分比
						minWidth : 20,//默认最小宽度
						maxWidth : null,
						align : 'left',
						_expand : false,//自定义列内容
						callBack : $.noop,
						hcls : '',//header cell自定义css 样式名
						bcls : '',//body cell自定义css 样式名
						fcls : '',//footer cell自定义css 样式名
						sortable : false, //需要引入sort.js
						fitColumn : true,//改变列大小
						casePerChange : true,//如果当前列设置的是百分比大小，手动修改列大小后 设置为true:grid刷新会失去百分比作用,false:grid刷新时不会失去百分比作用，而是失去手动修改后的宽度
						reader : {},//映射eg { 0:'男',1:'女',2:function(){return:'未知'},'_default_':'其他' }
						forceFit : true,//接受forceFit开启时自动设置列大小 checkbox edit 列会设置为false
						disabled : false//当前列不可用
						}
1. field String 当前列的命名
2. index String 当前列与data数据对应字段，默认等于filed
3. title String 列头信息
4. width Number,String 当前列宽度 默认120px ,可接受 百分比，eg:以下参数是合格的: 120,'120px','30%'
5. align String 居中方式
6. _expand String(模板) 自定义单元格内容显示模板
7. callBack Function 单元格回调
8. hcls String 列头单元格样式
9. bcls String 单元格样式
10. fcls String 无效
11. sortable Boolean 排序
13. fitColumn Boolean 当前列是否可改变大小
14. disabled Boolean 当前列是否无效
